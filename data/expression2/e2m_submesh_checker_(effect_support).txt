@name E2M Submesh Checker (effect support)
@persist Meshes:table MeshCount MeshIds:string Matches
@persist MaterialKey:string ExactMatching PrintAll A
@persist [LabelColor ErrorColor]:vector
@outputs EntOverride:entity
@model models/sprops/cuboids/height24/size_1/cube_24x24x24.mdl



if(first()){
    #[ 
        INSTRUCTIONS:
        1. Set your desired model and bodygroups below.
        2. Type your desired material below (e.g. "acrylic").
        3. Place this E2 on a prop.
        4. Copy and paste the printed array into your E2M code, or into the "E2M Submesh Visualizer" E2.
    ]#
    
    # CONFIGURATION
    Model = "models/spfox/rdc/budd_rdc-2_seats.mdl" # Model path.
        # Tip: You can right-click the model in the 'Q' menu and click "Copy to clipboard" to get the model path.
        
    MaterialKey = "rubberfloor" # Name or part of name of the material you're looking for
    ExactMatching = 0 # If set to 1, looks for exact matches only (use full material path, e.g. "models/proppertextures/acrylic")
    PrintAll = 0 # If set to 1, prints every submesh regardless of material (matches are still highlighted)
                    # Warning: This WILL flood your chat & console.
              
            
    holoCreate(0,entity():toWorld(vec(0)),vec(1),entity():toWorld(ang(0)),vec(255),Model)
    holoParent(0,entity())
    local E = holoEntity(0)
    EntOverride = E
    
    # SET BODYGROUPS HERE
        E:setBodygroup(1,1) # Copy and paste this as many times as you need.
            
            
            
    # Don't touch anything below.
            
            
                  
                    
    A = 1
    Meshes = table()
    MeshCount = 0
    Matches = 0
    MeshIds = "("
    
    LabelColor = vec(160,255,100)
    ErrorColor = vec(255,80,80)
    
    let End = function(){
        printColor(LabelColor, "Matches: ", vec(255), Matches + "/" + MeshCount)
        
        if(MeshIds:length() > 1){
            printColor(LabelColor, "E2M submesh array: ", vec(255), "array" + MeshIds:replaceRE(",$",")"))
        }
        elseif(MaterialKey:length() == 0){
            printColor(vec(255), "(No material key provided)")
        }
        elseif(ExactMatching){
            printColor(ErrorColor, "No submeshes with exact material \"" + MaterialKey + "\"")
        }
        else{
            printColor(ErrorColor, "No submeshes with material containing \"" + MaterialKey + "\"")
        }
    }
    let Print = function(){
        Material = Meshes[A,string]
        if(ExactMatching & MaterialKey == Material){
            printColor(vec(0,255,100), "  submesh " + A + ": \"" + Material + "\"")
            MeshIds += (A + ",")
            Matches++
        }
        elseif(!ExactMatching & Material:find(MaterialKey) & MaterialKey:length() > 0){
            printColor(vec(0,255,100), "  submesh " + A + ": \"" + Material + "\"")
            MeshIds += (A + ",")
            Matches++
        }
        elseif(PrintAll){
            printColor(vec(170), "  submesh " + A + ": \"" + Material + "\"")
        }
        
        A++
    }
    let Pause = function(){
        printColor(LabelColor, "[SUBMESH TABLE]")
        timer("print",0.05,MeshCount,Print)
        timer("finish",(0.05*MeshCount)+1,End)
    }
    
    BG = E:getBodygroupMask()
    Meshes = E:getSubmeshMaterials()
    MeshCount = Meshes:count()
    
    printColor(LabelColor, "[MODEL INFO]")
    printColor(LabelColor, "  Model path: ", vec(255), E:model())
    printColor(LabelColor, "  Bodygroup mask: ", vec(255), BG)
    printColor(LabelColor, "  Total submeshes: ", vec(255), MeshCount)
    
    
    timer(2,Pause)
}
